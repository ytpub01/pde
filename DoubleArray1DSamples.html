+++++++++++++++++++++++++++++++++++++
Using the DoubleArray1D class

Required Support File: DoubleArray1D.h
+++++++++++++++++++++++++++++++++++++ 

#include <iostream> 
using namespace std; 
 
#include "DoubleArray1D.h" // Using the DoubleArray1D Class 

int main()
{
    DoubleArray1D V(3);   // Declare DoubleArray1D class instances size 3
    DoubleArray1D W(3);   // by invoking the DoubleArray1D(long n) constructor.

      
    long size;
    size = V.getSize();  // Accssing the size data member of the DoubleArray1D instance

    V(0) = 1.0;          // Setting the vector entries by implicitly invoking the
    V(1) = 2.0;          // double& operator()(long i) member function.
    V(2) = 2.0;

    W = V;               // Assign V's values to W by implicitly invoking the member function
                         // void operator=(DoubleVector& V)
                         // 
    
    V += W;              // Set V = V + W by implicity invoking the member function 
                         // void operator+=(DoubleVector& V)

    cout << V << endl;   // Output the results 
  

    return 0;
}

and return types of a function.

#include <iostream> 
using namespace std; 
 
#include "DoubleArray1D.h" // Using the DoubleArray1D Class 

//
// Function that computes the dot product of two DoubleArray1D
// instances. The arguments are being passed by reference to
// avoid extraneous copying. 
//

double vectorDot(DoubleArray1D& V, DoubleArray1D& W) 
{
    long  i;
    long size = V.getSize();       // extract size of vectors 
    double dotProd = 0.0;          // compute the dot products       

    for(i = 0; i < size; i++)
    {
        dotProd += V(i)*W(i);
    }
 
    return dotProd;
}

//
// Function that negates the components of a DoubleArray1D and 
// returns the result as a DoubleArray1D. 
//

DoubleArray1D negateVector(DoubleArray1D& V)
{
    DoubleArray1D  R(V); // Make a copy of the input vector for the 
                         // return argument. This will automatically 
                         // initialize the values of R with those of V.

    long i;
    long size = V.getSize();

    for(i = 0; i < size; i++)  // negate elements
    {
    R(i) *= -1.0;
    } 

    return R;
}
